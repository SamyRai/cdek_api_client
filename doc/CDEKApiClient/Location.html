def postal_codes(city_code = nil, use_live_data: false)
  if use_live_data
    city_code ? @client.request('get', "location/postalcodes?code=#{city_code}") : @client.request('get', 'location/postalcodes')
  else
    read_data_from_file('postal_codes_mapping.json')
  end
end
